# List

<Documenter component={List}>

```jsx preview
const items = [
  'Foo',
  'Bar',
  'Baz',
];

const [value, setValue] = useState(null);

<List
  items={items}
  value={value}
  onChange={value => setValue(value)}
  multiple>

  {({ item }) => <span>{item}</span>}

</List>
```

</Documenter>

### Multiple selection list with disabled items

```jsx preview
const items = [
  'Foo',
  'Bar',
  'Baz',
];

const [value, setValue] = useState(null);

<List
  items={items}
  value={value}
  onChange={value => setValue(value)}
  isItemDisabled={item => item === 'Bar'}
  multiple>

  {({ item }) => <span>{item}</span>}

</List>
```
