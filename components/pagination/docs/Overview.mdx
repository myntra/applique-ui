### Using onChange and setting values

```preview
const [total] = React.useState(200)
const [size, setSize] = React.useState(10)
const [page, setPage] = React.useState(1)

return (
    <Pagination total={total} size={size} page={page} onChange={({page, size}) => { setSize(size); setPage(page) }}/>
)
```
